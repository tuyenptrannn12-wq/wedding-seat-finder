<script>

const guests = {

  "sarah nguyen": "Table 6",

  "john tran": "Table 3",

  "emily le": "Table 2"

};


function findGuest() {

  let input = document.getElementById("search").value.toLowerCase().trim();

  let result = document.getElementById("result");


  if (guests[input]) {

    result.innerHTML = "✨ You are seated at <br>" + guests[input];

  } else {

    result.innerHTML = "No match found. Please check spelling.";

  }

}
